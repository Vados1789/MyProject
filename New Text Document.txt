CREATE TABLE Logins (
    id INT PRIMARY KEY IDENTITY,
    profile_id INT,
    username VARCHAR(100) UNIQUE,
    password VARCHAR(255),
    FOREIGN KEY (profile_id) REFERENCES Profiles(id)
);
